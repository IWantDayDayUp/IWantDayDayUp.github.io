#include <bits/stdc++.h>

using namespace std;

constexpr int V = 1e5;
constexpr int inf = 1e9;

int pos[V + 1];
vector<int> factors[V + 1];

void solve()
{
    int n, q;
    cin >> n >> q;

    vector<int> a(n);
    for (int i = 0; i < n; i++)
    {
        cin >> a[i];
    }

    vector<int> k(q), r(q);
    vector<vector<int>> qry(n);
    for (int i = 0; i < q; i++)
    {
        int l;
        cin >> k[i] >> l >> r[i];
        l--;

        qry[l].push_back(i);
    }

    vector<long long> ans(q);

    vector<int> nxt(n, inf);
    for (int i = n - 1; i >= 0; i--)
    {
        nxt[i] = pos[a[i]];
        pos[a[i]] = i;
    }

    for (int l = 0; l < n; l++)
    {
        for (auto i : qry[l])
        {
            while (k[i] % a[l] == 0)
            {
                k[i] /= a[l];
            }
            int t = inf;
            for (auto d : factors[k[i]])
            {
                t = min(t, pos[d]);
            }

            if (t >= r[i])
            {
                ans[i] += 1ll * (r[i] - l) * k[i];
            }
            else
            {
                ans[i] += 1ll * (t - l) * k[i];
                assert(t > l);
                qry[t].push_back(i);
            }
        }

        pos[a[l]] = nxt[l];
    }

    for (int i = 0; i < q; i++)
    {
        cout << ans[i] << endl;
    }
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);

    fill(pos, pos + V + 1, inf);
    for (int i = 2; i <= V; i++)
    {
        for (int j = i; j <= V; j += i)
        {
            factors[j].push_back(i);
        }
    }

    int t;
    cin >> t;
    while (t--)
    {
        solve();
    }

    return 0;
}
