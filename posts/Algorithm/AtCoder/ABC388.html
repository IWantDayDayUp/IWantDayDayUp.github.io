📌 B - Heavy Snake

#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, D;
    cin >> N >> D;

    vector<int> T(N), L(N);
    for (int i = 0; i < N; i++)
    {
        cin >> T[i] >> L[i];
    }

    for (int d = 1; d <= D; d++)
    {
        int ans = 0;
        for (int i = 0; i < N; i++)
        {
            int cur = (L[i] + d) * T[i];
            ans = max(ans, cur);
        }

        cout << ans << endl;
    }

    return 0;
}

📌 D - Coming of Age Celebration

Tips

By focusing not on receiving but on giving stones, we can find the correct answer by interpreting the problem as follows:

  • Alien iii has AiA_iAi​ stones and will become an adult iii years later
  • Alien iii gives each of aliens i+1,i+2,...,Ni + 1, i + 2, ..., Ni+1,i+2,...,N a stone(if he has one) in this order when he becomes an adult
  • How many stones will they finally have?

Let sequence CCC of length NNN be the total number of stones that alien iii will recive

Suppose that alien iii will give aliens i+1,i+2,...,Rii + 1, i + 2, ..., R_ii+1,i+2,...,Ri​ after becoming an adult. Here, we want to add 111 to CjC_jCj​ for each j=i+1,i+2,...,Rij = i + 1, i + 2, ..., R_ij=i+1,i+2,...,Ri​, but naively doing so costs a total time complexity of O(N2)O(N^2)O(N2). Instead we use the cumulative sum trick to optimize it

Let sequence DDD of length N+1N + 1N+1 be the differential sequence of CCC, initially filled with 000:

  • For each i=1,2,...,Ni = 1, 2, ..., Ni=1,2,...,N, do the following:
    • if i>1i > 1i>1, let CiC_iCi​ be Ci−1+DiC_{i - 1} + D_iCi−1​+Di​. This determines the number of stones when alien iii becomes an adult, yielding RiR_iRi​
    • Add Di+1D_{i + 1}Di+1​ to 111 and DRi+1D_{R_i + 1}DRi​+1​ to −1-1−1

This procedure can be done in a total of O(N)O(N)O(N) time

#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<int> A(N);
    for (int i = 0; i < N; i++)
    {
        cin >> A[i];
    }

    vector<int> C(N), D(N + 1);

    for (int i = 0; i < N; i++)
    {
        if (i != 0)
        {
            C[i] = C[i - 1] + D[i];
            A[i] += C[i];
        }

        int cnt = min(N - i - 1, A[i]);
        A[i] -= cnt;

        D[i + 1] += 1;
        D[min(N, i + cnt + 1)] -= 1;
    }

    for (int i = 0; i < N; i++)
    {
        cout << A[i] << " \n"[i == N - 1];
    }

    return 0;
}

📌 E - Simultaneous Kagamimochi

#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    deque<int> q;

    for (int i = 0; i < N / 2; i++)
    {
        int a;
        cin >> a;

        q.push_back(a);
    }

    long long ans = 0;
    for (int i = N / 2; i < N; i++)
    {
        int a;
        cin >> a;

        if (q.front() * 2 <= a)
        {
            ans++;
            q.pop_front();
        }
    }

    cout << ans << endl;

    return 0;
}
