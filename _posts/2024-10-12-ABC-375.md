## E - 3 Team Division (DP)

Let $S = \sum_{i = 1}^N(B_i)$, and if $S$ is not a multiple of $3$, the answer is $-1$.

For $S$ is a multiple of $3$, since $S$ is small, we take `Dynamic Programming(DP)` approach.

Let `$dp_{i, x, y, z}$`: the minimum number of people who need to switch their teams so that people $1, 2, ..., i$ are assigned to each team and the strengths of team $1, 2, 3$ are $x, y, z$, respectively.

Naively performing this DP does not finish within the execution time limit. However, considering the sum of strengths of people $1, 2, ..., i$, it turns out that $z$ is uniquely determined given $i, x, y$. This allows ue not to maintain $z$ as the index.

So, let `$dp_{i, x, y}$`: the minimum number of people who need to switch their teams so that people $1, 2, ..., i$ are assigned to each team and the strengths of team $1, 2$ are $x, y$, respectively(if it is impossible, let it $\infin$).

The sought value is `$dp_{N, \frac{S}{3}, \frac{S}{3}}$`, so it is sufficient to support indices $x$ and $y$ up to $\frac{S}{3}$.(If you try to maintain up to $S$, it might not finish in time depending on other constant factors.)

Alternatively, since $1 \le B_i$, one can kick out the index $i$, which requires extra caution on the constant factor.

```cpp
#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    vector<int> a(n), b(n);
    int total = 0;
    for (int i = 0; i < n; i++)
    {
        cin >> a[i] >> b[i];

        total += b[i];
    }

    if (total % 3 != 0)
    {
        cout << -1 << endl;
        return 0;
    }

    int A = 501;
    int inf = 1e9;
    vector<vector<int>> dp(A, vector<int>(A, inf));
    dp[0][0] = 0;

    for (int i = 0; i < n; i++)
    {
        vector<vector<int>> ep(A, vector<int>(A, inf));

        int e = b[i];

        for (int j = 0; j < A; j++)
        {
            for (int k = 0; k < A; k++)
            {
                // assign to group 1
                if (j + e < A)
                    ep[j + e][k] = min(ep[j + e][k], dp[j][k] + (a[i] != 1));
                // assign to group 2
                if (k + e < A)
                    ep[j][k + e] = min(ep[j][k + e], dp[j][k] + (a[i] != 2));
                // assign to group 3
                ep[j][k] = min(ep[j][k], dp[j][k] + (a[i] != 3));
            }
        }

        dp = move(ep);
    }

    int res = dp[total / 3][total / 3];
    cout << (res == inf ? -1 : res) << endl;

    return 0;
}
```

## F - Road Blocked (reverse, Warshall-Floyd)

By processing the queries `in reverse order`, we now know how to handle `edge-adding` queries instead of removal queries.

If there is no edge-adding queries, one can do precalculation of `Warshall-Floyd algorithm` in a total of $O(N^3)$ time to answer each query in $O(1)$ time.

When an edge is added between vertices $u$ and $v$, the shortest path from vertex $u$ to vertex $v$ is always one of the following:

- the original path `x -> y`
- the path `x -> u -> v -> y`
- the path `x -> v -> v -> y`

If we know the original all-pair shortest distances, this can be checked in $O(1)$ time, for a total of $O(N^2)$ time to update the all-pair shortest distances.

```cpp
#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N, M, Q;
    cin >> N >> M >> Q;

    vector<int> A(M), B(M);
    vector<long long> C(M);
    for (int i = 0; i < M; i++)
    {
        cin >> A[i] >> B[i] >> C[i];
        A[i]--, B[i]--;
    }

    vector<bool> closed(M);
    vector<array<int, 2>> qry(Q);
    for (int i = 0; i < Q; i++)
    {
        int t;
        cin >> t;
        if (t == 1)
        {
            int j;
            cin >> j;
            j--;
            closed[j] = 1;
            qry[i] = {-1, j};
        }
        else
        {
            int a, b;
            cin >> a >> b;
            qry[i] = {a - 1, b - 1};
        }
    }

    vector dis(N, vector<long long>(N, 1e18));
    for (int i = 0; i < N; i++)
    {
        dis[i][i] = 0;
    }
    for (int i = 0; i < M; i++)
    {
        if (!closed[i])
        {
            dis[A[i]][B[i]] = min(dis[A[i]][B[i]], C[i]);
            dis[B[i]][A[i]] = min(dis[B[i]][A[i]], C[i]);
        }
    }

    // Warshall-Floyd algorithm
    for (int k = 0; k < N; k++)
    {
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                dis[i][j] = min(dis[i][j], dis[i][k] + dis[k][j]);
            }
        }
    }

    vector<long long> ans;
    for (int i = Q - 1; i >= 0; i--)
    {
        auto [a, b] = qry[i];
        if (a == -1)
        {
            for (int j = 0; j < N; j++)
            {
                for (int k = 0; k < N; k++)
                {
                    // update Warshall-Floyd algorithm
                    dis[j][k] = min(dis[j][k], dis[j][A[b]] + dis[k][B[b]] + C[b]);
                    dis[j][k] = min(dis[j][k], dis[j][B[b]] + dis[k][A[b]] + C[b]);
                }
            }
        }
        else
        {
            ans.push_back(dis[a][b] == 1e18 ? -1 : dis[a][b]);
        }
    }

    reverse(ans.begin(), ans.end());
    for (auto i : ans)
    {
        cout << i << endl;
    }

    return 0;
}
```
